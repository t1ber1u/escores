import dotenv from 'dotenv';

dotenv.config();

class EnvHelper {
  static getEnvVariable(key: string, defaultValue?: string): string {
    const value = process.env[key];
    if (!value) {
      if (defaultValue !== undefined) {
        console.warn(`Warning: Environment variable ${key} is not set or has no value. Using default value: ${defaultValue}`);
        return defaultValue;
      } else {
        throw new Error(`Environment variable ${key} is not set and no default value is provided.`);
      }
    }
    return value;
  }
}

export default EnvHelper;
